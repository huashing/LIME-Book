 57   /** @class バイト列から任意のバイト/ビット列を取得するための内部クラス @inner
 58    *  @name  BinaryTable-ByteHandler
 59    *  @param {Array<Number>} [bytes] バイト列の各バイト値を整数値表現した配列
 60    */
 61   function ByteHandler(bytes) {
 62     /**#@+ @memberOf BinaryTable-ByteHandler.prototype */
 63     /** 対象バイト列の各バイト単位を整数値表現した配列． @propertyEx  BinaryTable-ByteHandler.prototype.bytes  |Array<Number>|new Array() */
 64     this.bytes   = bytes || [];
 65     /** バイト列内の現在のバイト位置．最左が0．          @propertyEx  BinaryTable-ByteHandler.prototype.byteIdx|Number|0 */
 66     this.byteIdx = 0;
 67     /** 現在のバイト位置におけるビット位置．最左が0．    @propertyEx  BinaryTable-ByteHandler.prototype.bitPos |Number|0 */
 68     this.bitPos  = 0;
 69     /**#@-*/
 70   }
 71   /**#@+ @methodOf BinaryTable-ByteHandler */
 72   /** 保持するバイト列の最後尾に指定のバイト値を追加する．
 73    *  @name    BinaryTable-ByteHandler.prototype.push
 74    *  @param   {Number} b 追加するバイト値の整数値表現(0〜255)
 75    */
 76   ByteHandler.prototype.push = function(b) {
 77     this.bytes.push(b);
 78   };
 79   /** バイト列から次に取得するビット値が存在するかを判定する．
 80    *  @name    BinaryTable-ByteHandler.prototype.hasNextBit
 81    *  @returns {Boolean} ビット値が存在有無(有=true)
 82    */
 83   ByteHandler.prototype.hasNextBit = function() {
 84     var max = this.bytes.length - 1;
 85     return( (this.byteIdx <  max) ||
 86            ((this.byteIdx == max) && (this.bitPos < 8)));
 87   };
 88   /** バイト列から次に取得するバイト値が存在するかを判定する．
 89    *  @name    BinaryTable-ByteHandler.prototype.hasNextByte
 90    *  @returns {Boolean} バイト値が存在有無(有=true)
 91    */
 92   ByteHandler.prototype.hasNextByte = function() {
 93     var max = this.bytes.length - 1;
 94     return( (this.byteIdx <  max) ||
 95            ((this.byteIdx == max) && (this.bitPos === 0)));
 96   };
 97   /** バイト列から指定の長さのバイト列を，各バイト単位の整数値表現を格納した配列として取得する．
 98    *  @name    BinaryTable-ByteHandler.prototype.getBytes
 99    *  @param   {Number}        len バイト長
100    *  @returns {Array<Number>} 各バイト単位の整数値表現を格納する配列
101    */
102   ByteHandler.prototype.getBytes = function(len) {
103     len = isNaN(len) ? 1 : len;
104 
105     var ret = [], i;
106     var bytes = this.bytes, idx = this.byteIdx, pos = this.bitPos;
107     if (pos === 0) {
108       for(i = 0; i < len; i++) {
109         ret[i] = (bytes[idx++] || 0x00);
110       }
111     } else {
112       for(i = 0; i < len; i++) {
113         var buf = ((bytes[idx++] || 0x00) << 8) | (bytes[idx] || 0x00);
114         ret[i] = (buf >> (8 - pos)) & 0xff;
115       }
116     }
117     this.byteIdx = idx;
118     
119     return(ret);
120   };
121   /** バイト列から1バイト値の整数値表現を取得する．
122    *  @name    BinaryTable-ByteHandler.prototype.getByte
123    *  @returns {Number} バイト値の整数値表現
124    */
125   ByteHandler.prototype.getByte = function() {
126     return(this.getBytes(1)[0]);
127   };
128   /** バイト列から指定の長さのバイト列を，符号付き整数値として取得する．
129    *  @name    BinaryTable-ByteHandler.prototype.getBytesAsInteger
130    *  @param   {Number} len バイト数
131    *  @returns {Number}     バイト値の符号付き整数値表現
132    */
133   ByteHandler.prototype.getBytesAsInteger = function(len) {
134     var bytes = this.getBytesAsUnsignedInteger(len);
135 
136     var msb = 0x80;
137     for(var i = 1; i < len; i++) {
138       msb  = (msb  << 8) | 0x00;
139     }
140     // 最上位ビットチェック(負数チェック)
141     if ((ret & msb) > 0) ret = (ret & ~msb) - msb;
142     
143     return(ret);
144   };
145   /** バイト列から指定の長さのバイト列を，符号なし整数値として取得する．
146    *  @name    BinaryTable-ByteHandler.prototype.getBytesAsUnsignedInteger
147    *  @param   {Number} len バイト数
148    *  @returns {Number}     バイト値の符号なし整数値表現
149    */
150   ByteHandler.prototype.getBytesAsUnsignedInteger = function(len) {
151     var bytes = this.getBytes(len);
152     var ret = 0;
153     for(var i = 0; i < len; i++) {
154       ret = (ret << 8) | (bytes[i] || 0x00);
155     }
156     return(ret);
157   };
158   /** バイト列からビット列を符号なし整数値表現で取得する．
159    *  @name    BinaryTable-ByteHandler.prototype.getBits
160    *  @param   {Number} len    ビット長
161    *  @param   {Number} [mask] ビット長のマスク表現
162    *  @returns {Number}        ビット列の符号なし整数値表現
163    */
164   ByteHandler.prototype.getBits = function(len, mask) {
165     var i, l;
166     if (!mask) {
167       for(mask = 0x01, i = 1; i < len; i++) { mask = (mask << 1) | 0x01; }
168     }
169   
170     var ret, shift;
171     var bytes = this.bytes, idx = this.byteIdx, pos = this.bitPos;
172     if (len < 8) { // 1byte未満なら
173       shift = 8 - pos - len;
174       if (shift >= 0) {
175         ret = (bytes[idx] || 0x00);
176         ret = (ret >>> shift) & mask;
177         this.bitPos += len;
178       } else {
179         ret = ((bytes[idx++] || 0x00) << 8) & (bytes[idx] || 0x00);
180         ret = (ret >>> (8 + shift)) & mask;
181         this.pos = -shift;
182       }
183       this.byteIdx = idx;
184     } else { // 1byte以上なら
185       ret = (bytes[idx] || 0x00);
186       for(i = 1, l = Math.ceil((pos + len) / 8); i < l; i++) {
187         ret = (ret << 8) & (bytes[++idx] || 0x00);
188       }
189       this.bitPos  = (pos + len) % 8;
190       shift        = (8 - this.bitPos) % 8;
191       ret          = (ret >>> shift) & mask;
192       this.byteIdx = idx;
193     }
194     return(ret);
195   };
196   /** バイト列からビット列を符号なし整数値表現で取得する．
197    *  @name    BinaryTable-ByteHandler.prototype.getBitsAsUnsignedInteger
198    *  @param   {Number} len    ビット長
199    *  @param   {Number} [mask] ビット長のマスク表現
200    *  @returns {Number}        ビット列の符号なし整数値表現
201    */
202   ByteHandler.prototype.getBitsAsUnsignedInteger = function(len, mask) {
203     return(this.getBits(len, mask));
204   };
205   /** バイト列からビット列を符号付き整数値表現で取得する．<br>
206    *  ビット長の最上位ビットを符号ビットとして判定し，バイトアライメントは考慮しない．
207    *  @name    BinaryTable-ByteHandler.prototype.getBitsAsInteger
208    *  @param   {Number} len    ビット長
209    *  @param   {Number} [mask] ビット長のマスク表現
210    *  @returns {Number}        ビット列の符号付き整数値表現
211    */
212   ByteHandler.prototype.getBitsAsInteger = function(len) {
213     var msb = 0x01;
214     for(i = 1; i < len; i++) {
215       msb  = (msb  << 1) | 0x00;
216     }
217     var mask = ~msb;
218     
219     var bits = this.getBits(len, msb | mask);
220     // 最上位ビットチェック(負数チェック)
221     if ((bits & msb) > 0) bits = (bits & mask) - msb;
222     
223     return(bits);
224   };

227   /**#@+ @methodOf BinaryTable @inner @static @description (無名関数内でのユーティリティ関数であり，スコープ外からは見えない．)<br> */
228   /** BinaryTableのフォーマットを解析する．
229    *  @name    BinaryTable.parseStructure
230    *  @param   {String}        structure     BinaryTableのフォーマットを記述した文字列
231    *  @returns {Array<Object>}               次の値をプロパティに持つオブジェクトの集合<br>
232    *                                         [.type]    フィールドの型(B/U/I/S/Z/P)
233    *                                         [.length]  フィールドの長さ
234    *                                         [.unit]    フィールドの長さを表す単位(B/b/V)
235    *                                         [.mask]    ビット長のマスク表現(unitがb(ビット)の場合)
236    *  @throw   {BinaryTableInvalidStructure} フォーマットの記述が正しくない場合
237    */
238   function parseStructure(structure) {
239     var columns    = structure.split(',');
240     var lengthByte = columns.shift();
241 
242     structure = [];
243     var length, unit, mask;
244     for(i = 0, l = columns.length; i < l; i++) {
245       var match = /^(\w):(\d+)(\w)$/.exec(columns[i]);
246       if (!match) throw('[BinaryTableInvalidStructure] : '+columns[i]);
247 
248       
249       length = parseInt(match[2], 10);
250       unit   = match[3];
251       mask   = 0x01;
252       
253       if (unit == 'b') {
254         for(i = 1; i < length; i++) {
255           mask = ((mask << 1) | 0x01);
256         }
257       }
258       structure.push({
259         type   : match[1],
260         length : length,
261         unit   : unit,
262         mask   : mask
263       });
264     }
265     structure.lengthByte = lengthByte;
266     return(structure);
267   }
268   /** 郵便番号を整数値表現に変換する．
269    *  @name    BinaryTable.zipCodeToInteger
270    *  @param   {Number} digit 郵便番号の上位2桁
271    *  @param   {Number} a     郵便番号の上位3桁目
272    *  @param   {Number} b     郵便番号の下位1桁目
273    *  @param   {Number} c     郵便番号の下位2桁目
274    *  @param   {Number} d     郵便番号の下位3桁目
275    *  @param   {Number} e     郵便番号の下位4桁目
276    *  @returns {Number}       郵便番号の整数値表現
277    */
278   function zipCodeToInteger(digit, a, b, c, d, e) {
279     return(digit * 100000 + a * 10000 + b * 1000 + c * 100 * d * 10 * e);
280   }
281   /** 郵便番号の整数値表現がlist内に含まれているかを判定する．
282    *  @name    BinaryTable.includeZipCode
283    *  @param   {Number}        zipCode 郵便番号の整数値表現
284    *  @param   {Array<Object>} list    次の値を持つオブジェクトのリスト<br>
285    *                                   [.from] 範囲の下限値
286    *                                   [.to]   範囲の上限値
287    *  @returns {Boolean}               包含有無(有=true)
288    */
289   function includeZipCode(zipCode, list) {
290     for(var i = 0, l = list.length; i < l; i++) {
291       var condition = list[i];
292       if ((condition.from <= zipCode) && (zipCode <= condition.to)) return(true);
293     }
294     return(false);
295   }

313   // BinaryTableの実装
314   // @param {Sting} table_ref 表ファイルの指定
315   // @param {Sting} structure 表のフォーマット指定
316   // @throw {BinaryTableInvalidStructure} フォーマットの記述が正しくない場合
317   function BinaryTableClass(table_ref, structure) {
318     // ARIB STD-B24 第二編「7.5.2.2 BinaryTable オブジェクトのコンストラクタ」
319     // データの取得
320     var uri = BML.Util.combinePath(table_ref, BML.uri);
321     var ajax = new BML.Ajax(uri, {
322     overrideMimeType : 'text/plain; charset=x-user-defined',
323     asynchronous     : false,
324     method           : 'GET'
325     });
326     // @Todo：ステータスコードの扱いがゾンザイ
327     if (ajax.response.statusCode != 200) throw('[FileNotFound] :'+uri);
328 
329     // バイト列としてバッファ
330     var byteHandler = new ByteHandler(), i, l;
331     var stream = ajax.response.responseText;
332     for(i = 0, l = stream.length; i < l; i++) {
333       byteHandler.push(stream._charCodeAt_(i) & 0x00ff);
334     }
335     
336     // structureの解析
337     structure = parseStructure(structure);
338     
339     // テーブルの作成
340     var table = [], entry;
341     while(1) {
342       if (!byteHandler.hasNextBit()) break;
343       // @Todo：lengthByteを数えていないが問題ないか?
344       
345       entry = [];
346       for(i = 0, l = structure.length; i < l; i++) {
347         var field = structure[i];
348         
349         switch(field.type) {
350           case('B') : {
351             // Boolean
352             entry.push(new Boolean((byteHandler.getBits(1) > 0)));
353           } break;
354           case('U') : {
355             // Unsigned Integer
356             entry.push(new Number((field.unit == 'b') ?
357                                   byteHandler.getBitsAsUnsignedInteger (field.length, field.mask) :
358                                   byteHandler.getBytesAsUnsignedInteger(field.length)));
359             } break;
360           case('I') : {
361             entry.push(new Number(byteHandler.getBytesAsInteger(field.length)));
362           } break;
363           case('S') : {
364             if (field.unit == 'B') { // 固定長で文字列を取得する
365               entry.push(String.charCodeFrom.apply(null,
366                                                       byteHandler.getBytes(field.length)));
367             } else { // 'V'：可変長→文字列の順番で取得する
368               length = byteHandler.getBytesAsUnsignedInteger(field.length);
369               entry.push(String.charCodeFrom.apply(null,
370                                                    byteHandler.getBytes(length)));
371             }
372           } break;
373           case('P') : {
374             (field.unit == 'b') ?
375               byteHandler.getBitsAsUnsignedInteger (field.length, field.mask) :
376               byteHandler.getBytesAsUnsignedInteger(field.length);
377           } break;
378           case('Z') : {
379             length    = byteHandler.getBytesAsUnsignedInteger(field.length);
380             var whole = { include : {}, exclude : {} };
381 
382             var includeListNum = byteHandler.getBytesAsUnsignedInteger(1) - 1;
383             var excludeListNum = byteHandler.getBytesAsUnsignedInteger(1) / 4;
384             includeListNum = includeListNum / 4 - excludeListNum;
385 
386             BML.Util.each.call([[whole.exclude, excludeListNum],
387                                 [whole.include, includeListNum]], function(param) {
388               var list = param[0], rangeFrom, msb, from, to, digits, flag, a, b, c, d, e;
389               for(var i = 0, l = param[1];  i < l; i++) {
390                 msb = byteHandler.getBits(1);
391 
392                 if (!msb) {
393                   from = byteHandler.getBitsAsUnsignedInteger(7) * 10;
394                   msb  = byteHandler.getBits(1);
395                   to   = byteHandler.getBitsAsUnsignedInteger(7) * 10;
396                   list.push({
397                     from  : (from + 0) * 10000 +     0,
398                     to    : (to   + 9) * 10000 +  9999
399                   });
400                   if (msb) {
401                     from = byteHandler.getBitsAsUnsignedInteger(7) * 10;
402                     msb  = byteHandler.getBits(1);
403                     to   = byteHandler.getBitsAsUnsignedInteger(7) * 10;
404                     list.push({
405                       from  : (from + 0) * 10000 +     0,
406                       to    : (to   + 9) * 10000 +  9999
407                     });
408                   } else {
409                     byteHandler.getBytesAsInteger(2);
410                   }
411 
412                 } else {
413                   digits = byteHandler.getBitsAsUnsignedInteger(7);
414                   flag   = byteHandler.getBitsAsUnsignedInteger(4);
415                   a      = byteHandler.getBitsAsUnsignedInteger(4);
416                   b      = byteHandler.getBitsAsUnsignedInteger(4);
417                   c      = byteHandler.getBitsAsUnsignedInteger(4);
418                   d      = byteHandler.getBitsAsUnsignedInteger(4);
419                   e      = byteHandler.getBitsAsUnsignedInteger(4);
420                   
421                   switch(flag) {
422                     case(0x08) : {  // 3digit list
423                       BML.Util.each.call([a, b, c, d, e], function(num) {
424                         if (num == 0x0f) return;
425                         list.push({
426                           from  : zipCodeToInteger(digits, num, 0, 0, 0, 0),
427                           to    : zipCodeToInteger(digits, num, 9, 9, 9, 9)
428                         });
429                       });
430                       rangeFrom = null;
431                     } break;
432                     case(0x09) : { // 3digit range
433                       BML.Util.each.call([[b, c], [d, e]], function(num) {
434                         if ((num[0] == 0x0f) || (num[1] == 0x0f)) return;
435                         list.push({
436                           from  : zipCodeToInteger(digits, num[0], 0, 0, 0, 0),
437                           to    : zipCodeToInteger(digits, num[1], 9, 9, 9, 9)
438                         });
439                       });
440                       rangeFrom = null;
441                     } break;
442                     case(0x0a) : { // 5digit list
443                       BML.Util.each.call([[b, c], [d, e]], function(num) {
444                         if ((num[0] == 0x0f) || (num[1] == 0x0f)) return;
445                         list.push({
446                           from  : zipCodeToInteger(digits, a, num[0], num[1], 0, 0),
447                           to    : zipCodeToInteger(digits, a, num[0], num[1], 9, 9)
448                         });
449                       });
450                       rangeFrom = null;
451                     } break;
452                     case(0x0b) : { // 5digit range from
453                       rangeFrom = zipCodeToInteger(digits, a, b, c, 0, 0);
454                     } break;
455                     case(0x0c) : { // 5digit range to
456                       if (!rangeFrom) throw('[InvalidFieldFormat] : 5 digit range from not defined');
457                       list.push({
458                         from  : rangeFrom,
459                         to    : zipCodeToInteger(digits, a, b, c, 9, 9)
460                       });
461                       rangeFrom = null;
462                     } break;
463                     case(0x0d) : { // 7digit range from
464                       rangeFrom = zipCodeToInteger(digits, a, b, c, d, e);
465                     } break;
466                     case(0x0e) : { // 7digit range to
467                       if (!rangeFrom) throw('[InvalidFieldFormat] : 7 digit range from not defined');
468                       list.push({
469                         from  : rangeFrom,
470                         to    : zipCodeToInteger(digits, a, b, c, d, e)
471                       });
472                       rangeFrom = null;
473                     } break;
474                     case(0x0f) : { // 7digit list
475                       list.push({
476                         from  : zipCodeToInteger(digits, a, b, c, d, e),
477                         to    : zipCodeToInteger(digits, a, b, c, d, e)
478                       });
479                       rangeFrom = null;
480                     } break;
481                     default : { break; }
482                   }
483                 }
484               }
485             });
486             entry.push(whole);
487           } break;
488         default : { break; }
489         }
490       }
491       table.push(entry);
492     }
493     var buf = []; // remove padding field
494     for(i = 0, l = structure.length; i < l; i++) {
495       entry = structure[i];
496       if (entry.type != 'P') buf.push(entry);

591    *  @param   {Object}  compared       比較の対象(String/Number/Boolean)

595    *  @param   {Array}   resultArray    出力レコードを格納する配列
596    *  @returns {Number}                 連続するレコードを格納した配列
597    *  @throw   {InvalidArguments}       引数が正しくない場合
598    */
599   BinaryTableClass.prototype.search = function() {
600     var args = arguments.slice(0);
601 
602     var startRow    = args.shift();
603     var resultArray = args.pop();
604     var limitCount  = args.pop();
605     var logic       = args.pop();
606     if (args.length % 3) throw('[InvalidArguments] : not enough arguments');
607 
608     var conditionList = [];
609     for(var i = 0, l = args.length / 3; i < l; i++) {
610       conditionList.push({
611         searchedColumn : args[i * 3 + 0],
612         compared       : args[i * 3 + 1],
613         operator       : args[i * 3 + 2]
614       });
615     }
616 
617     for(i = startRow, l = this.nrow; i < l; i++) {
618       if (limitCount <= 0) break;
619 
620       var row = this.table[i], flag = true;
621       for(var j = 0, m = conditionList.length; j < m; j++) {
622         var condition = conditionList[j];
623         var value     = row[condition.searchedColumn];
624         var compared  = condition.compared;
625 
626         switch(condition.operator) {
627           // Unsigned Integer or Signed Integer
628           case( 0) : { flag = value == compared; } break;
629           case( 1) : { flag = value != compared; } break;
630           case( 2) : { flag = value <  compared; } break;
631           case( 3) : { flag = value <= compared; } break;
632           case( 4) : { flag = value >  compared; } break;
633           case( 5) : { flag = value >= compared; } break;
634           case( 6) : { flag = value &  compared; } break;
635           case( 7) : { flag = value |  compared; } break;
636           case( 8) : { flag = value ^  compared; } break;
637           case( 9) : { flag = value & ~compared; } break;
638           case(10) : { flag = value | ~compared; } break;
639           case(11) : { flag = value ^ ~compared; } break;
640           // String
641           case(32) : { flag = value == compared;             } break;
642           case(33) : { flag = value.indexOf(compared) >=  0; } break;
643           case(34) : { flag = value.indexOf(compared) === 0; } break;
644           case(35) : { flag =
645             value.lastIndexOf(compared) == value.length - compared.length; } break;
646           case(36) : { flag = value != compared;             } break;
647           case(37) : { flag = value.indexOf(compared) <   0; } break;
648           // Boolean
649           case(64) : { flag = value == compared; } break;
650           case(65) : { flag = value != compared; } break;
651           // ZipCode
652           case(96) : { flag =  includeZipCode(compared, value.include) &&
653                               !includeZipCode(compared, value.exclude); } break;
654           case(97) : { flag = !includeZipCode(compared, value.include) &&
655                                includeZipCode(compared, value.exclude); } break;
656           default  : { flag = true; } break;
657         }

659         if (logic == flag) break;
660       }

663       limitCount--;
664     }
