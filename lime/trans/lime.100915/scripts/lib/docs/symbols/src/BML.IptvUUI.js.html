263   function EpgGetIntEventInfo(eit_type, event_id, description_type, param) {
264     if(isNaN(event_id) || (event_id < 0)) return(UIE_EPG_ERR_INVALID_EVENT_ID);
265     if(isNaN(description_type))           return(UIE_EPG_ERR_NG);
266 
267     var name = [iptv_uui.EpgType, iptv_uui.EpgNetworkId, iptv_uui.EpgServiceId].join('.') +
268                '_' + event_id + GET_EVENT_INFO_URL_SUFFIX;
269 
270     var target = getEventInfoCache[name];
271     if (!target) {
272       target = browser.transmitTextDataOverIP(GET_EVENT_INFO_URL_PREFIX + name, null, 'EUC-JP');
273       if (isNaN(target[0]) || (target[0] < 0)) return(UIE_EPG_ERR_NG);
274       target = ((target[2].split("\r\n"))[0]).split("\t");
275       getEventInfoCache[name] = target;
276     }
277 
278     var date = this.objectCreateTime;
279     date = new Date(date.getFullYear(), date.getMonth(), date.getDate(),date.getHours(),0,0,0);
280 
281     if      (description_type == EIT_START_TIME)       { return(Math.floor(browser.addDate(date, parseInt(target[0],10), 1).getTime() / 1000));}
282     else if (description_type == EIT_DURATION)         { return(parseInt(target[1],10)); }
283     else if (description_type == EIT_PARENTAL_RATING)  { return(parseInt(target[2],10)); }
284     else if (description_type == EIT_DESCRIPTORS)      { return(parseInt(target[3],10)); }
285     else if (description_type == EIT_CONTENT)          { return(parseInt(target[4],10)); }
286     else if (description_type == EIT_GROUP_TYPE)       { return(parseInt(target[5],10)); }
287     else if (description_type == EIT_GROUP_NETWORK_ID) { return(parseInt(target[6],10)); }
288     else if (description_type == EIT_GROUP_SERVICE_ID) { return(parseInt(target[7],10)); }
289     else if (description_type == EIT_GROUP_EVENT_ID)   { return(parseInt(target[8],10)); }
290     else if (description_type == EIT_FREE_CA_MODE)     { return(parseInt(target[9],10)); }
291     else                                               { return(UIE_EPG_ERR_NG);         }
292   }
