 75     ms = Math.max(100, Math.floor((ms + 99) / 100) * 100); // 設定値は100ms単位
 76     if (repeat === 0) repeat = -1;
 77     var timer = window.setInterval(function() {

 79       if ((repeat > 0) && (--repeat === 0)) window.clearInterval(timer);
 80     }, ms);

107   function formatNumber(value) {
108     value = String(value);
109     var len = value.length;
110     if (len < 3) return(value);
111     
112     var odd = len % 3, buf = [];
113     if (odd > 0) buf.push(value.substring(0, odd));
114     for(var i = odd; i < len; i+=3) {
115       buf.push(value.substring(i, i+3));
116     }
117     return(buf.join(','));
118   }
